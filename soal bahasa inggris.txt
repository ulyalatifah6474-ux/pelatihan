    <p>Seret dan jatuhkan kata-kata di bawah ini ke kotak untuk membentuk kalimat yang benar.</p>

    <div id="word-bank"></div>

    <textarea id="sentence" placeholder="Taruh kata di sini dengan urutan yang benar..." ondragover="allowDrop(event)" ondrop="drop(event)"></textarea>

  <script>
    const words = ["I", "am", "going", "to", "the", "market"];
    let shuffled = [];

    function shuffle(array) {
      let currentIndex = array.length, randomIndex;
      while (currentIndex !== 0) {
        randomIndex = Math.floor(Math.random() * currentIndex);
        currentIndex--;
        [array[currentIndex], array[randomIndex]] = [
          array[randomIndex], array[currentIndex]
        ];
      }
      return array;
    }

    function loadWords() {
      const bank = document.getElementById("word-bank");
      bank.innerHTML = "";
      shuffled = shuffle([...words]);
      shuffled.forEach(word => {
        const div = document.createElement("div");
        div.textContent = word;
        div.classList.add("word");
        div.draggable = true;
        div.id = "word-" + word;
        div.addEventListener("dragstart", drag);
        bank.appendChild(div);
      });
    }

    function drag(event) {
      event.dataTransfer.setData("text", event.target.id);
    }

    function allowDrop(event) {
      event.preventDefault();
    }

    function drop(event) {
      event.preventDefault();
      const data = event.dataTransfer.getData("text");
      const wordElement = document.getElementById(data);
      if (wordElement) {
        const textarea = document.getElementById("sentence");
        textarea.value += (textarea.value ? "-" : "") + wordElement.textContent;
        wordElement.remove(); // hilangkan kata yang sudah dipindah
      }
    }

    function cekHasil() {
      const answer = document.getElementById("sentence").value.trim();
      const feedback = document.getElementById("feedback");
      if (answer === "I-am-going-to-the-market") {
        feedback.textContent = "Jawaban Anda benar, hebat!";
        feedback.style.color = "green";
      } else {
        feedback.textContent = "Maaf, jawaban Anda salah. Ulangi lagi.";
        feedback.style.color = "red";
      }
    }

    function resetGame() {
      document.getElementById("sentence").value = "";
      document.getElementById("feedback").textContent = "";
      loadWords();
    }

    // Inisialisasi saat pertama kali dibuka
    loadWords();
  </script>
